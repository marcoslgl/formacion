1.Añadido el return para que la funcion devuelva los datos
import { ref } from "vue";
export function useCounter() {
  const count = ref(0);
  function increment() {
    count.value++;
  }
  return {count, increment};
}

2.Si se recorre el array por indice y se elimina el primer elemento vue lo detecta como el ultimo 
<script setup lang="ts">
import { ref } from "vue";
const items = ref(["Apple", "Banana", "Cherry"]);
</script>

<template>
  <ul>
    <li v-for="item in items" :key="item">
      {{ item }}
    </li>
  </ul>
</template>

3.Añadido ref para que el valor counter se pueda modificar
<script setup lang="ts">
import { ref } from "vue";
const counter = ref(0);
</script>

<template>
  <p v-if="counter === 10">Número, {{ counter }}</p>
  <p v-else>El contador es {{ counter }}</p>
</template>
